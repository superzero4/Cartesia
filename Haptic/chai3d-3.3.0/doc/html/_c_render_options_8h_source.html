136inline bool SECTION_RENDER_PARTS_WITH_MATERIALS(cRenderOptions a_options, bool a_useTransparency)
137{
138 return (!(
139 ((a_options.m_render_opaque_objects_only) && (a_useTransparency)) ||
140 ((a_options.m_render_transparent_front_faces_only || a_options.m_render_transparent_back_faces_only) && (!a_useTransparency))
141 ));
142}

154inline bool SECTION_RENDER_OPAQUE_PARTS_ONLY(cRenderOptions a_options)
155{
156 return(!(a_options.m_render_transparent_back_faces_only ||
157 a_options.m_render_transparent_front_faces_only));
158}
