540inline double cModuloRange(double a_value, double a_min, double a_max)
541{
542 double range = a_max - a_min;
543 double result = a_value;
544
545 if (a_value > a_max)
546 {
547 double value = a_value - a_min;
548 double num = std::trunc(value / range);
549 result = a_value - num * range;
550 }
551 else if (a_value < a_min)
552 {
553 double value = a_max - a_value;
554 double num = std::trunc(value / range);
555 result = a_value + num * range;
556 }
557
558 return result;
559}

736inline int cNumDigits(int a_value)
737{
738 int digits = 0;
739
740 if (a_value == 0) { return (1); }
741
742 double value = (double)(abs(a_value));
743 while (value >= 1.0)
744 {
745 value /= 10;
746 digits++;
747 }
748 return (digits);
749}

1069bool inline cEqualPoints(const cVector3d& a_point1,
1070 const cVector3d& a_point2,
1071 const double a_epsilon = C_SMALL)
1072{
1073 // accelerated path for exact equality
1074 if (a_epsilon == 0.0)
1075 {
1076 if ((a_point1(0) == a_point2(0)) && (a_point1(1) == a_point2(1)) && (a_point1(2) == a_point2(2)))
1077 {
1078 return (true);
1079 }
1080 else
1081 {
1082 return (false);
1083 }
1084 }
1085
1086 if ((fabs(a_point1(0) - a_point2(0)) < a_epsilon) &&
1087 (fabs(a_point1(1) - a_point2(1)) < a_epsilon) &&
1088 (fabs(a_point1(2) - a_point2(2)) < a_epsilon))
1089 {
1090 return (true);
1091 }
1092 else
1093 {
1094 return (false);
1095 }
1096}

1449inline double cAngle(const cVector3d& a_vector1,
1450 const cVector3d& a_vector2)
1451{
1452 // compute length of vectors
1453 double n1 = a_vector1.length();
1454 double n2 = a_vector2.length();
1455 double val = n1 * n2;
1456
1457 // check if lengths of vectors are not zero
1458 if (fabs(val) < C_SMALL)
1459 {
1460 return (0.0);
1461 }
1462
1463 // compute angle
1464 double result = a_vector1.dot(a_vector2)/(val);
1465 if (result > 1.0) { result = 1.0; }
1466 else if (result < -1.0) { result = -1.0; }
1467
1468 return(acos(result));
1469}
