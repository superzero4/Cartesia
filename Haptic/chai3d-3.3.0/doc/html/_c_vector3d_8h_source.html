307 inline bool set(const char* a_initStr)
308 {
309 // sanity check
310 if (a_initStr == 0) return (false);
311
312 // look for a valid-format string. ignore leading whitespace and ('s
313 const char* curpos = a_initStr;
314 while( (*curpos != '\0') &&
315 (*curpos == ' ' || *curpos == '\t' || *curpos == '('))
316 {
317 curpos++;
318 }
319
320 // parse data
321 double ax, ay, az;
322 std::string str = a_initStr;
323 std::locale loc;
324 for (int j=0; j<(int)(str.length()); j++) if (!std::isdigit(str[j], loc) && str[j] != '.') str[j] = ' ';
325 std::istringstream is(str);
326 if (!(is >> ax && !is.fail())) return (false);
327 if (!(is >> ay && !is.fail())) return (false);
328 if (!(is >> az && !is.fail())) return (false);
329
330 // copy the values we found
331 (*this)(0) = ax;
332 (*this)(1) = ay;
333 (*this)(2) = az;
334
335 // return result
336 return (true);
337 }
