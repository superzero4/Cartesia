496//------------------------------------------------------------------------------
497// OPERTAOR - (IMPLEMENTED FOR EDGE SORTING PURPOSES)
498//------------------------------------------------------------------------------
499inline bool operator<(const cEdge& a_edge0, const cEdge& a_edge1)
500{
501 double tol = 0.0;
502
503 cVector3d edge0Vertex0 = a_edge0.m_parent->m_vertices->getLocalPos(a_edge0.m_vertex0);
504 cVector3d edge0Vertex1 = a_edge0.m_parent->m_vertices->getLocalPos(a_edge0.m_vertex1);
505 cVector3d edge1Vertex0 = a_edge1.m_parent->m_vertices->getLocalPos(a_edge1.m_vertex0);
506 cVector3d edge1Vertex1 = a_edge1.m_parent->m_vertices->getLocalPos(a_edge1.m_vertex1);
507
508
509 if (cAbs(edge0Vertex0.x() - edge1Vertex0.x()) <= tol)
510 {
511 if (cAbs(edge0Vertex0.y() - edge1Vertex0.y()) <= tol)
512 {
513 if (cAbs(edge0Vertex0.z() - edge1Vertex0.z()) <= tol)
514 {
515 if (cAbs(edge0Vertex1.x() - edge1Vertex1.x()) <= tol)
516 {
517 if (cAbs(edge0Vertex1.y() - edge1Vertex1.y()) <= tol)
518 {
519 return (edge0Vertex1.z() < edge1Vertex1.z());
520 }
521 else
522 {
523 return (edge0Vertex1.y() < edge1Vertex1.y());
524 }
525 }
526 else
527 {
528 return (edge0Vertex1.x() < edge1Vertex1.x());
529 }
530 }
531 else
532 {
533 return (edge0Vertex0.z() < edge1Vertex0.z());
534 }
535 }
536 else
537 {
538 return (edge0Vertex0.y() < edge1Vertex0.y());
539 }
540 }
541 else
542 {
543 return (edge0Vertex0.x() < edge1Vertex0.x());
544 }
545}
