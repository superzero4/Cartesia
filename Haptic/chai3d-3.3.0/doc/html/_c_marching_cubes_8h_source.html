102inline cVector3d cVertexInterpolation(double a_isolevel,
103 cVector3d a_p1,
104 cVector3d a_p2,
105 double a_valp1,
106 double a_valp2)
107{
108 double mu;
109 cVector3d p;
110
111 if (cAbs(a_isolevel - a_valp1) < 0.00001)
112 return(a_p1);
113 if (cAbs(a_isolevel - a_valp2) < 0.00001)
114 return(a_p2);
115 if (cAbs(a_valp1 - a_valp2) < 0.00001)
116 return(a_p1);
117 mu = (a_isolevel - a_valp1) / (a_valp2 - a_valp1);
118
119 p(0) = a_p1(0) + mu * (a_p2(0) - a_p1(0));
120 p(1) = a_p1(1) + mu * (a_p2(1) - a_p1(1));
121 p(2) = a_p1(2) + mu * (a_p2(2) - a_p1(2));
122
123 return(p);
124}
