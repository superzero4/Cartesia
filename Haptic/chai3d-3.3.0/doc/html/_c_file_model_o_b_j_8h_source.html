105struct vertexIndexSet
106{
107 int vIndex;
108 int nIndex;
109 int tIndex;
110
111 vertexIndexSet()
112 {
113 vIndex = nIndex = tIndex = 0;
114 }
115 vertexIndexSet(int a_vIndex, int a_nIndex, int a_tIndex)
116 {
117 vIndex = a_vIndex;
118 nIndex = a_nIndex;
119 tIndex = a_tIndex;
120 }
121 vertexIndexSet(int a_vIndex)
122 {
123 vIndex = a_vIndex;
124 nIndex = 0;
125 tIndex = 0;
126 }
127};
128
129//------------------------------------------------------------------------------
130
131struct ltVertexIndexSet
132{
133 bool operator()(vertexIndexSet v1, vertexIndexSet v2) const
134 {
135 if (v1.vIndex < v2.vIndex) return 1;
136 if (v2.vIndex < v1.vIndex) return 0;
137 if (!g_objLoaderUnifyVerticesWithSamePosition)
138 {
139 if (v1.nIndex < v2.nIndex) return 1;
140 if (v2.nIndex < v1.nIndex) return 0;
141 if (v1.tIndex < v2.tIndex) return 1;
142 }
143 return 0;
144 }
145};
